#version 420

in vec3 vertex_position, vertex_normal;
in vec2 vertex_texture;

out vec3 position_eye, normal_eye;
out vec2 texture_coord;

uniform vec4 LightPosition; // = vec4 (0.0, 10.0, -5.0, 1.0); // Light position in world coords.Make this a uniform so you can change position of "sun" in code?
out vec3 light_position_eye;

uniform mat4 view;
uniform mat4 proj;
uniform mat4 model;

void main(){

  position_eye = vec3(view*model*vec4(vertex_position, 1.0));
  normal_eye = vec3(view*model*vec4(vertex_normal, 0.0));
  light_position_eye = vec3(view * LightPosition);
     
  texture_coord = vertex_texture;
  gl_Position = proj* vec4(position_eye, 1.0);
}
